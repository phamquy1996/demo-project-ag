objOrderPanel.WK_UpdateOrderPanel = function (res) { // console.log(res, 'e.data'); // Chèn đúng vị trí theo pk_orderNo giảm dần (mới nhất trên cùng) const newPk = parseInt(res.pk_orderNo) || 0; const $newRow = $(res.data); let inserted = false; let $lastDataRow = null; $('#tbodyContent tr').each(function () { const $this = $(this); // Lấy pk_orderNo từ element có id bắt đầu bằng "no_" const rowPkId = $this.find('[id^="no_"]').attr('id'); if (rowPkId) { const rowPk = parseInt(rowPkId.replace('no_', '')) || 0; if (newPk > rowPk) { // Chèn trước row có pk nhỏ hơn $this.before($newRow); inserted = true; return false; // break $.each } $lastDataRow = $this; // Theo dõi row data cuối cùng } }); if (!inserted) { if ($lastDataRow) { // newPk nhỏ hơn tất cả data rows → chèn sau row data cuối $lastDataRow.after($newRow); } else { // Bảng trống hoặc chỉ có placeholder → prepend lên đầu $('#tbodyContent').prepend($newRow); } } objOrderPanel.updateMax(res.pk_orderNo); let rows = $('#tbodyContent tr'); if (rows.length > 6) { rows.slice(6).remove(); } if (keyOrder.size > 20) { keyOrder = new Set([...keyOrder].slice(0, 20)); } if (orderQueueMissingData.length > 0) { // Xóa timer cũ nếu nó đang tồn tại (clearTimeout) if (timeoutOrderMissing) { clearTimeout(timeoutOrderMissing); } // Thiết lập timer mới (setTimeout) // Ví dụ trì hoãn 500ms trước khi thực thi hàm cập nhật timeoutOrderMissing = setTimeout(() => { objOrderPanel.updateRowOrderRealtimeDataMissing(); // Sau khi chạy xong có thể xóa ID để reset trạng thái timeoutOrderMissing = null; }, 200); } // console.log(res.pk_orderNo, "res.pk_orderNo") }
